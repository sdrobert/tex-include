\newglossaryentry{cint}{ %
  name={closed interval}, %
  symbol={\ensuremath{\left[x, y\right]}}, %
  description={ %
    The ordered set of all values between $x$ and $y$, including boundaries.
    $\{a : x \leq a \leq y\}$. %
  } %
}

\newglossaryentry{field}{ %
  name={field}, %
  description={ %
    A set endowed with addition and multiplication; their identities ($0$ and
    $1$); inverses (subtraction and division, the latter not defined for $0$);
    commutativity; associativity; and distributivity. %
  } %
}
\newglossaryentry{ghomo}{ %
  name={group homomorphism}, %
  description={ %
    A map $\phi$ from one group $\set{G}$ to another $\set{H}$ preserving group
    structure: $\phi(gg') = \phi(g)\phi(g')$. %
  } %
}
\newglossaryentry{group}{ %
  name={group}, %
  description={ %
    A set and a closed, binary operation on said set, which has an identity, an
    inverse, and is associative. %
  } %
}
